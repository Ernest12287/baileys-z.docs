// Validate critical configuration
function validateConfig() {
    const errors = [];

    // Validate bot prefix
    if (botConfig.BOT.PREFIX.length > 2) {
        errors.push('Bot prefix should be 1-2 characters max');
    }

    // Validate owner number format
    const ownerNumber = botConfig.BOT.OWNER_NUMBER;
    if (ownerNumber && !/^\d+$/.test(ownerNumber.replace('+', ''))) {
        errors.push('Owner number must contain only digits (with optional + prefix)');
    }

    // Validate rate limits
    if (botConfig.PERFORMANCE.RATE_LIMIT < 1) {
        errors.push('Rate limit must be at least 1 message per minute');
    }

    // Validate message delays
    if (botConfig.PERFORMANCE.MESSAGE_DELAY < 100) {
        errors.push('Message delay should be at least 100ms to avoid being flagged as spam');
    }

    // Return validation result
    return {
        isValid: errors.length === 0,
        errors: errors
    };
}

// Perform configuration validation
const validation = validateConfig();
if (!validation.isValid) {
    console.warn('‚ö†Ô∏è  Configuration warnings:');
    validation.errors.forEach(error => console.warn(`   - ${error}`));
}

// üéØ CONFIGURATION UTILITIES

// Check if a number is allowed to interact with the bot
function isNumberAllowed(phoneNumber) {
    const cleanNumber = phoneNumber.replace('+', '').replace(/\s/g, '');
    
    // Check blocked list first (highest priority)
    if (botConfig.SECURITY.BLOCKED_NUMBERS.includes(cleanNumber)) {
        return false;
    }
    
    // If strict mode is enabled, check allowed list
    if (botConfig.SECURITY.STRICT_MODE) {
        return botConfig.SECURITY.ALLOWED_NUMBERS.includes(cleanNumber);
    }
    
    // Otherwise, allow all numbers not explicitly blocked
    return true;
}

// Get environment name (development, production, etc.)
function getEnvironment() {
    return process.env.NODE_ENV || 'development';
}

// Check if running in production
function isProduction() {
    return getEnvironment() === 'production';
}

// Check if running in development
function isDevelopment() {
    return getEnvironment() === 'development';
}

    getEnvironment,
    isProduction,
    isDevelopment

2Ô∏è‚É£ Configuration Structure: Organized by Purpose

üèóÔ∏è Logical Grouping Strategy

BOT - Identity & Behavior

Your bot's personality and basic settings:

  • NAME: How your bot introduces itself
  • PREFIX: The character that triggers commands (!help, !status)
  • OWNER_NUMBER: Your phone number for admin privileges
  • AUTO_REPLY: Enable/disable automatic responses

Environment variable pattern: process.env.BOT_NAME || 'MyWhatsAppBot'

This means: "Use the environment variable if it exists, otherwise use this default value."

SECURITY - Access Control

Who can interact with your bot:

  • ALLOWED_NUMBERS: Whitelist of approved numbers
  • BLOCKED_NUMBERS: Blacklist of banned numbers
  • STRICT_MODE: Only allow numbers in the whitelist

Pro tip: Use .split(',') to convert comma-separated strings into arrays.

PERFORMANCE - Rate Limiting

Avoiding spam detection and bans:

  • MESSAGE_DELAY: Minimum time between messages
  • RATE_LIMIT: Maximum messages per minute
  • MAX_MESSAGE_LENGTH: Prevent overly long messages

Critical: These settings help your bot behave like a human, reducing ban risk.

CONNECTION - Session Management

How your bot connects to WhatsApp:

  • AUTH_METHOD: QR code or pairing code
  • BROWSER: Browser fingerprint to avoid detection
  • MAX_RECONNECT_ATTEMPTS: Self-healing limits
FEATURES - Module Toggles

Enable/disable functionality:

  • GROUPS: Work in group chats
  • BROADCAST: Send messages to multiple users
  • AUTO_DELETE: Self-cleaning messages

Why this matters: Quickly disable features without deleting code.

SERVICES - External APIs

Integration with other services:

  • OPENAI_API_KEY: For AI-powered responses
  • WEATHER_API_KEY: For weather commands
  • WEBHOOK_URL: For external notifications

Security note: These should ALWAYS use environment variables, never hardcoded values.

üîí Environment Variable Best Practices

‚úÖ DO:
  • Add .env to your .gitignore file
  • Use descriptive, consistent naming (BOT_NAME, not name)
  • Provide sensible default values in config.js
  • Use different .env files for development/production
‚ùå DON'T:
  • Commit API keys or sensitive data to version control
  • Use environment variables for code logic (use them for configuration only)
  • Create overly complex nested environment variable names
  • Forget to document available environment variables

// Rate limiting utility
import config from './config.js';

const messageTimestamps = new Map();

function isRateLimited(phoneNumber) {
    const now = Date.now();
    const oneMinuteAgo = now - 60000;
    
    // Get recent messages for this number
    const recentMessages = (messageTimestamps.get(phoneNumber) || [])
        .filter(timestamp => timestamp > oneMinuteAgo);
    
    // Check if over limit
    if (recentMessages.length >= config.PERFORMANCE.RATE_LIMIT) {
        return true;
    }
    
    // Add current message
    recentMessages.push(now);
    messageTimestamps.set(phoneNumber, recentMessages);
    return false;
}
